112  virtual void coreWorkerThread(size_t) {
113  std::chrono::steady_clock::time_point callTime;
114  std::shared_ptr<TimerTask> timerTask;
115  while(runStateOf(ctl_.load()) <= SHUTDOWN) {
116  if (timerTasks_.empty()) {
117  std::unique_lock<std::mutex> lk(mutex_);
118  notEmpty_.wait(lk);
119  }
120  {
121  std::lock_guard<std::mutex> lock(mutex_);
122  timerTask = timerTasks_.front();
123  std::pop_heap(timerTasks_.begin(), timerTasks_.end(), Comp());
124  timerTasks_.pop_back();
125  }
126  //如果是fixed rate就在执行前更新下次执行时间
127  //否则在执行后更新
128  callTime = timerTask->callTime_;
129  if (timerTask->fixedRate_) {
130  timerTask->callTime_ = std::chrono::steady_clock::now() + timerTask->interval_;
131  std::this_thread::sleep_until(callTime);
132  timerTask->operator()();
133  } else {
134  std::this_thread::sleep_until(callTime);
135  timerTask->operator()();
136  timerTask->callTime_ = std::chrono::steady_clock::now() + timerTask->interval_;
137  }
138  //执行完毕并更新时间完成,将任务放回小顶堆
139  {
140  std::lock_guard<std::mutex> lock_guard(mutex_);
141  timerTasks_.push_back(timerTask);
142  std::push_heap(timerTasks_.begin(), timerTasks_.end(), Comp());
143  }
144  }
