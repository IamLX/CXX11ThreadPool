109  std::chrono::steady_clock::time_point callTime;
110  std::shared_ptr<TimerTask> timerTask;
111  while(runStateOf(ctl_.load()) <= SHUTDOWN) {
112  if (timerTasks_.empty()) {
113  std::unique_lock<std::mutex> lk(mutex_);
114  notEmpty_.wait(lk);
115  }
116  {
117  std::lock_guard<std::mutex> lock(mutex_);
118  timerTask = timerTasks_.front();
119  std::pop_heap(timerTasks_.begin(), timerTasks_.end(), Comp());
120  timerTasks_.pop_back();
121  }
122  //如果是fixed rate就在执行前更新下次执行时间
123  //否则在执行后更新
124  callTime = timerTask->callTime_;
125  if (timerTask->fixedRate_) {
126  timerTask->callTime_ = std::chrono::steady_clock::now() + timerTask->interval_;
127  std::this_thread::sleep_until(callTime);
128  timerTask->operator()();
129  } else {
130  std::this_thread::sleep_until(callTime);
131  timerTask->operator()();
132  timerTask->callTime_ = std::chrono::steady_clock::now() + timerTask->interval_;
133  }
134  //执行完毕并更新时间完成,将任务放回小顶堆
135  {
136  std::lock_guard<std::mutex> lock_guard(mutex_);
137  timerTasks_.push_back(timerTask);
138  std::push_heap(timerTasks_.begin(), timerTasks_.end(), Comp());
139  }
140  }
